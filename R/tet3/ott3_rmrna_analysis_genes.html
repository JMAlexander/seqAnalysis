Extract FPKM values from Cuffdiff output

cd <- read.delim("~/s2/data/rna/cuffdiff/omp_ott3_rmrna/gene_exp.diff")
cd.fc2 <- cd[abs(cd$log2.fold_change.) >= 2, ]
cd.fc2.fpkm1 <- cd.fc2[cd.fc2$value_1 >= 1 | cd.fc2$value_2 >= 1, ]
