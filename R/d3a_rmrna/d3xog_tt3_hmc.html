hmc.m <- ddply(hmc.m, .(id), mutate, value.norm.wt = (value + 0.01)/(value[variable == 
    "wt"] + 0.01))

gg <- ggplot(hmc.m, aes(value, color = variable))
gg + geom_density() + scale_color_brewer(palette = "Set1") + coord_cartesian(xlim = c(0, 
    4))
