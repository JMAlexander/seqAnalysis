mod <- namerows(mod)

## Using id as id variables

v <- str_split(mod.m$variable, "_")
mod.m$mod <- factor(unlist(lapply(v, function(x) x[1])))
mod.m$geno <- factor(unlist(lapply(v, function(x) x[2])), levels = c("wt", "het", 
    "ko"))
